local function ResolveItemName(inputName)
    if ItemsFolder:FindFirstChild(inputName) then return inputName end
    for _, item in pairs(ItemsFolder:GetChildren()) do
        local dn = item:FindFirstChild("DisplayName")
        if dn and string.lower(dn.Value) == string.lower(inputName) then return item.Name end
        if string.lower(item.Name) == string.lower(inputName) then return item.Name end
    end
    return inputName
end

local function AnimateBar(success)
    local steps = 20
    local delay = 1 / steps  -- total 1 segundo de animação
    for i = 1, steps do
        local pct = i / steps
        TweenService:Create(BarFill, TweenInfo.new(delay * 0.9), {
            Size = UDim2.new(pct, 0, 1, 0)
        }):Play()
        BarPct.Text = math.floor(pct * 100) .. "%"
        if i == 10 then
            LoadSub.Text = "Returning..."
        end
        task.wait(delay)
    end
    BarPct.Text = "100%"
    if success then
        LoadResult.TextColor3 = Color3.fromRGB(60, 200, 60)
        LoadResult.Text = "Success! Returned to previous position."
    else
        LoadResult.TextColor3 = Color3.fromRGB(220, 60, 60)
        LoadResult.Text = "Failed! Target not found."
    end
    task.wait(1.4)
    LoadingFrame.Visible = false
end

local function FindNearestDollHouse()
    local nearest, nearestDist = nil, math.huge
    local function Search(parent, depth)
        if depth > 5 then return end
        for _, obj in pairs(parent:GetChildren()) do
            if obj.Name == "Doll House" then
                local part = obj:FindFirstChildWhichIsA("BasePart") or obj.PrimaryPart
                if part then
                    local d = (RootPart.Position - part.Position).Magnitude
                    if d < nearestDist then nearestDist = d nearest = obj end
                end
            end
            Search(obj, depth + 1)
        end
    end
    Search(WS, 0)
    return nearest
end

local function ClickDoll(doll)
    local cd = doll:FindFirstChildWhichIsA("ClickDetector", true)
    if cd then fireclickdetector(cd) return true end
    return false
end

                -- Tenta teleportar
                local success = pcall(function()
                    RootPart.CFrame = TARGET_CFRAME
                end)

                if success then
                    -- Anima barra até 50% enquanto espera 1s no alvo
                    local halfSteps = 10
                    for i = 1, halfSteps do
                        local pct = i / (halfSteps * 2)
                        TweenService:Create(BarFill, TweenInfo.new(0.05), {
                            Size = UDim2.new(pct, 0, 1, 0)
                        }):Play()
                        BarPct.Text = math.floor(pct * 100) .. "%"
                        task.wait(0.05)
                    end

                    LoadSub.Text = "Waiting at target..."
                    task.wait(0.5)

                    -- Volta para posição anterior
                    RootPart.CFrame = savedPos
                    LoadSub.Text = "Returning..."

                    -- Anima barra de 50% a 100%
                    for i = halfSteps + 1, halfSteps * 2 do
                        local pct = i / (halfSteps * 2)
                        TweenService:Create(BarFill, TweenInfo.new(0.05), {
                            Size = UDim2.new(pct, 0, 1, 0)
                        }):Play()
                        BarPct.Text = math.floor(pct * 100) .. "%"
                        task.wait(0.05)
                    end

                    BarPct.Text = "100%"
                    LoadResult.TextColor3 = Color3.fromRGB(60, 200, 60)
                    LoadResult.Text = "Success! Returned to previous position."
                else
                    -- Falha
                    BarFill.Size = UDim2.new(1, 0, 1, 0)
                    BarFill.BackgroundColor3 = Color3.fromRGB(200, 50, 50)
                    BarPct.Text = "Error"
                    LoadResult.TextColor3 = Color3.fromRGB(220, 60, 60)
                    LoadResult.Text = "Failed! Could not teleport."
                    task.wait(1.5)
                    LoadingFrame.Visible = false
                    autoGoldActive = false
                    break
                end

                task.wait(1.2)
                LoadingFrame.Visible = false

                if autoGoldActive then
                    task.wait(0.3)
                end
            end
            SetStatus("Auto Gold stopped.", Color3.fromRGB(70,70,70))
        end)
    else
        LoadingFrame.Visible = false
        SetStatus("Auto Gold stopped.", Color3.fromRGB(70,70,70))
    end
end)

local function PlaceObjectAt(obj, targetCFrame)
    -- Pega todas as BaseParts
    local parts = {}
    for _, d in pairs(obj:GetDescendants()) do
        if d:IsA("BasePart") then
            table.insert(parts, d)
        end
    end
    -- Também inclui se o próprio objeto for BasePart
    if obj:IsA("BasePart") then table.insert(parts, obj) end

    if #parts == 0 then return end

    -- Calcula centro atual
    local center = Vector3.new(0, 0, 0)
    for _, p in pairs(parts) do
        center = center + p.Position
    end
    center = center / #parts

    -- Offset para a nova posição
    local offset = targetCFrame.Position - center

    for _, p in pairs(parts) do
        p.Anchored = true
        p.CanCollide = false
        p.Locked = false
        p.CastShadow = false
        -- Mantém rotação original, apenas translada
        p.CFrame = CFrame.new(p.Position + offset) * (p.CFrame - p.CFrame.Position)
    end
end
