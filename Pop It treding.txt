local function ResolveItemName(inputName)
    if ItemsFolder:FindFirstChild(inputName) then return inputName end
    for _, item in pairs(ItemsFolder:GetChildren()) do
        local dn = item:FindFirstChild("DisplayName")
        if dn and string.lower(dn.Value) == string.lower(inputName) then return item.Name end
        if string.lower(item.Name) == string.lower(inputName) then return item.Name end
    end
    return inputName
end

local function FindNearestDollHouse()
    local nearest, nearestDist = nil, math.huge
    local function Search(parent, depth)
        if depth > 4 then return end
        for _, obj in pairs(parent:GetChildren()) do
            if obj.Name == "Doll House" then
                local part = obj:FindFirstChildWhichIsA("BasePart") or obj.PrimaryPart
                if part then
                    local d = (RootPart.Position - part.Position).Magnitude
                    if d < nearestDist then nearestDist = d nearest = obj end
                end
            end
            Search(obj, depth + 1)
        end
    end
    Search(WS, 0)
    return nearest
end

local function ClickDoll(doll)
    local cd = doll:FindFirstChildWhichIsA("ClickDetector", true)
    if cd then fireclickdetector(cd) return true end
    return false
end

    autoGoldActive = state
    if state then
        SetStatus("Auto Gold ativo...", Color3.fromRGB(255, 200, 0))
        task.spawn(function()

                -- Salva posição atual antes de ir
                lastPos = RootPart.CFrame

                -- Teleporta para o Board
                local ok, board = pcall(function()
                    return WS.Boards.Board.MAIN
                end)
                if ok and board then
                    local part = board:FindFirstChildWhichIsA("BasePart") or board.PrimaryPart
                    if part then
                        RootPart.CFrame = part.CFrame + Vector3.new(0, 4, 0)
                    end
                else
                    SetStatus("Board não encontrado!", Color3.fromRGB(220,60,60))
                    task.wait(2)
                    autoGoldActive = false
                    break
                end

                -- Espera 1 segundo no board
                task.wait(1)

                -- Volta para a posição anterior
                if lastPos then
                    RootPart.CFrame = lastPos
                end

                task.wait(0.5)
            end
            SetStatus("Auto Gold parado.", Color3.fromRGB(80,80,80))
        end)
    else
        -- Se desligou, volta imediatamente
        if lastPos then
            RootPart.CFrame = lastPos
        end
        SetStatus("Auto Gold parado.", Color3.fromRGB(80,80,80))
    end
end)

    local standsFolder = RS:FindFirstChild("Stands")
    if standsFolder then
        for i, standName in ipairs(STAND_NAMES) do
            local stand = standsFolder:FindFirstChild(standName)
            if stand then
                local clone = stand:Clone()
                clone.Name = standName
                clone.Parent = WorldFolder

                -- Posiciona os filhos no mundo
                local posX = BASE_X - 200 + (i-1) * SPACING
                local posZ = BASE_Z - 40

                -- Move todas as BaseParts do clone
                local parts = {}
                for _, desc in pairs(clone:GetDescendants()) do
                    if desc:IsA("BasePart") then
                        table.insert(parts, desc)
                    end
                end

                -- Calcula centro atual das parts
                if #parts > 0 then
                    local center = Vector3.new(0,0,0)
                    for _, p in pairs(parts) do
                        center = center + p.Position
                    end
                    center = center / #parts

                    local offset = Vector3.new(posX, BASE_Y + 5, posZ) - center
                    for _, p in pairs(parts) do
                        p.Anchored = true
                        p.CanCollide = false
                        p.CFrame = p.CFrame + offset
                    end
                end

                -- Label com nome
                local labelPart = Instance.new("Part")
                labelPart.Size = Vector3.new(1, 1, 1)
                labelPart.Position = Vector3.new(posX, BASE_Y + 12, posZ)
                labelPart.Anchored = true
                labelPart.Transparency = 1
                labelPart.CanCollide = false
                labelPart.Parent = WorldFolder

                local bg = Instance.new("BillboardGui", labelPart)
                bg.Size = UDim2.new(0, 120, 0, 25)
                bg.StudsOffset = Vector3.new(0, 2, 0)
                bg.AlwaysOnTop = true
                local txt = Instance.new("TextLabel", bg)
                txt.Size = UDim2.new(1,0,1,0)
                txt.BackgroundTransparency = 1
                txt.Text = standName
                txt.TextColor3 = Color3.fromRGB(255, 220, 100)
                txt.TextSize = 12
                txt.Font = Enum.Font.GothamBold
                txt.TextWrapped = true

                table.insert(clonedObjects, clone)
                table.insert(clonedObjects, labelPart)
            end
        end
    end

    -- Clona NPCs (linha de baixo)
    local npcsFolder = RS:FindFirstChild("NPCs")
    if npcsFolder then
        for i, npcName in ipairs(NPC_NAMES) do
            local npc = npcsFolder:FindFirstChild(npcName)
            if npc then
                local clone = npc:Clone()
                clone.Name = npcName
                clone.Parent = WorldFolder

                local posX = BASE_X - 100 + (i-1) * SPACING
                local posZ = BASE_Z + 50

                -- Torna visível e ancora
                local parts = {}
                for _, desc in pairs(clone:GetDescendants()) do
                    if desc:IsA("BasePart") then
                        desc.Transparency = 0
                        desc.Anchored = true
                        desc.CanCollide = false
                        table.insert(parts, desc)
                    end
                    -- Torna visível MeshParts, SpecialMesh etc
                    if desc:IsA("SpecialMesh") then
                        -- nada, só garantir que está no clone
                    end
                    -- Habilita humanoid se tiver
                    if desc:IsA("Humanoid") then
                        desc.DisplayDistanceType = Enum.HumanoidDisplayDistanceType.None
                    end
                end

                -- Centraliza
                if #parts > 0 then
                    local center = Vector3.new(0,0,0)
                    for _, p in pairs(parts) do
                        center = center + p.Position
                    end
                    center = center / #parts
                    local offset = Vector3.new(posX, BASE_Y + 5, posZ) - center
                    for _, p in pairs(parts) do
                        p.CFrame = p.CFrame + offset
                    end
                end

                -- Label nome
                local labelPart = Instance.new("Part")
                labelPart.Size = Vector3.new(1,1,1)
                labelPart.Position = Vector3.new(posX, BASE_Y + 14, posZ)
                labelPart.Anchored = true
                labelPart.Transparency = 1
                labelPart.CanCollide = false
                labelPart.Parent = WorldFolder

                local bg = Instance.new("BillboardGui", labelPart)
                bg.Size = UDim2.new(0, 120, 0, 25)
                bg.StudsOffset = Vector3.new(0, 2, 0)
                bg.AlwaysOnTop = true
                local txt = Instance.new("TextLabel", bg)
                txt.Size = UDim2.new(1,0,1,0)
                txt.BackgroundTransparency = 1
                txt.Text = npcName
                txt.TextColor3 = Color3.fromRGB(100, 200, 255)
                txt.TextSize = 12
                txt.Font = Enum.Font.GothamBold
                txt.TextWrapped = true

                table.insert(clonedObjects, clone)
                table.insert(clonedObjects, labelPart)
            end
        end
    end

    -- Teleporta jogador para o mundo visual
    local savedPos = RootPart.CFrame
    RootPart.CFrame = CFrame.new(BASE_X, BASE_Y + 6, BASE_Z - 100)

    return savedPos
end
